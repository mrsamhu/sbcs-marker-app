    <script>
        async function submitAnswer() {
            const btn = document.getElementById('submitBtn');
            const feedbackBox = document.getElementById('feedbackBox');
            
            const supportText = document.getElementById('supportBox').value.trim();
            const rejectText = document.getElementById('rejectBox').value.trim();
            const evalText = document.getElementById('evalBox').value.trim();

            const fullAnswer = `SUPPORT:\n${supportText}\n\nREJECT:\n${rejectText}\n\nEVALUATION:\n${evalText}`;

            if (supportText === "" && rejectText === "") {
                alert("Please write at least one paragraph before submitting!");
                return;
            }

            btn.innerText = "🤖 Reading your essay...";
            btn.disabled = true;
            feedbackBox.style.display = "none";
            feedbackBox.className = ""; 

            try {
                // Call our secure Vercel backend instead of Google directly
                const response = await fetch('/api/grade', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ essay: fullAnswer })
                });

                const data = await response.json();

                if (!response.ok) throw new Error(data.error || 'Server error');
                
                feedbackBox.innerHTML = "<strong>AI Examiner Report:</strong><br><br>" + data.feedback.replace(/\n/g, '<br>');
                feedbackBox.style.display = "block";

            } catch (error) {
                feedbackBox.innerHTML = "<strong>Error:</strong> " + error.message;
                feedbackBox.classList.add("error-box");
                feedbackBox.style.display = "block";
            } finally {
                btn.innerText = "Submit to AI Examiner";
                btn.disabled = false;
            }
        }
    </script>
